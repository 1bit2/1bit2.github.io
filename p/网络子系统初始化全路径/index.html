网络子系统初始化全路径

Linux 5.15.78 网络协议栈的启动初始化过程：从 start_kernel() 到网络就绪的完整流程。

一、初始化总体架构

Linux 网络子系统的初始化遵循分层、分阶段的设计原则，通过 initcall 机制按依赖顺序执行。

1.1 initcall 层次体系

// net/socket.c
static int __init sock_init(void)
{
    int err;

    /* 步骤1: 初始化网络 sysctl 基础设施
     * 创建 /proc/sys/net 目录，后续协议在此下创建配置项
     * 例如: /proc/sys/net/ipv4/tcp_* */
    err = net_sysctl_init();
    if (err)
        goto out;

    /* 步骤2: 初始化 sk_buff slab 缓存
     * sk_buff 是网络数据包的核心结构
     * 每个网络包都用 sk_buff 表示，高频分配/释放 */
    skb_init();

    /* 步骤3: 初始化 socket inode 缓存
     * socket 在内核中表现为特殊文件，需要 inode */
    init_inodecache();

    /* 步骤4: 注册 socket 伪文件系统 (sockfs)
     * socket 以文件形式暴露给用户空间
     * socket()/accept() 返回的 fd 指向 sockfs 中的文件 */
    err = register_filesystem(&sock_fs_type);
    if (err)
        goto out;
    /* 挂载 sockfs，获取 vfsmount 供后续 socket 创建使用 */
    sock_mnt = kern_mount(&sock_fs_type);
    if (IS_ERR(sock_mnt)) {
        err = PTR_ERR(sock_mnt);
        goto out_mount;
    }

    /* 步骤5: 初始化 netfilter 框架
     * 为 iptables/nftables 等提供 hook 点
     * 注意：真正的协议初始化在后续 initcall 中 */
#ifdef CONFIG_NETFILTER
    err = netfilter_init();
    if (err)
        goto out;
#endif

    /* 初始化 PTP 协议分类器 */
    ptp_classifier_init();

out:
    return err;

out_mount:
    unregister_filesystem(&sock_fs_type);
    goto out;
}

core_initcall(sock_init);

2.2 关键设计

  • sk_buff 缓存池: 网络数据包的核心数据结构，通过 slab 分配器高效管理
  • sockfs 伪文件系统: 使 socket 可以通过文件描述符访问，统一 VFS 接口
  • netfilter 框架: 为后续防火墙功能奠定基础，注册 hook 点

// net/core/dev.c
static int __init net_dev_init(void)
{
    int i, rc = -ENOMEM;

    BUG_ON(!dev_boot_phase);

    /* 步骤1: 初始化 /proc/net 设备统计文件 */
    if (dev_proc_init())
        goto out;

    /* 步骤2: 初始化网络设备 kobject 框架
     * 为网络设备在 sysfs 中创建目录 */
    if (netdev_kobject_init())
        goto out;

    /* 步骤3: 初始化协议类型哈希表
     * ptype_all: 所有协议类型的链表
     * ptype_base[PTYPE_HASH_SIZE]: 按协议号哈希的链表
     * 用于收包时根据协议号分发到对应处理函数 */
    INIT_LIST_HEAD(&ptype_all);
    for (i = 0; i < PTYPE_HASH_SIZE; i++)
        INIT_LIST_HEAD(&ptype_base[i]);

    INIT_LIST_HEAD(&offload_base);

    /* 步骤4: 注册网络命名空间子系统操作 */
    if (register_pernet_subsys(&netdev_net_ops))
        goto out;

    /* 步骤5: 初始化每个 CPU 的收包队列
     * 使用 per-CPU 设计避免锁竞争 */
    for_each_possible_cpu(i) {
        struct work_struct *flush = per_cpu_ptr(&flush_works, i);
        struct softnet_data *sd = &per_cpu(softnet_data, i);

        INIT_WORK(flush, flush_backlog);

        /* 初始化输入包队列 */
        skb_queue_head_init(&sd->input_pkt_queue);
        skb_queue_head_init(&sd->process_queue);

        /* 初始化 NAPI 轮询列表 */
        INIT_LIST_HEAD(&sd->poll_list);
        sd->output_queue_tailp = &sd->output_queue;

        /* 初始化 backlog 设备（用于软中断收包） */
        init_gro_hash(&sd->backlog);
        sd->backlog.poll = process_backlog;
        sd->backlog.weight = weight_p;
    }

    dev_boot_phase = 0;

    /* 步骤6: 注册 loopback 环回设备
     * loopback 设备是第一个注册的网络设备 */
    if (register_pernet_device(&loopback_net_ops))
        goto out;

    if (register_pernet_device(&default_device_ops))
        goto out;

    /* 步骤7: 注册网络收发软中断处理函数
     * NET_TX_SOFTIRQ: 发送软中断
     * NET_RX_SOFTIRQ: 接收软中断 */
    open_softirq(NET_TX_SOFTIRQ, net_tx_action);
    open_softirq(NET_RX_SOFTIRQ, net_rx_action);

    /* 步骤8: 注册 CPU 热插拔回调
     * 当 CPU 下线时处理该 CPU 的网络队列 */
    rc = cpuhp_setup_state_nocalls(CPUHP_NET_DEV_DEAD, "net/dev:dead",
                       NULL, dev_cpu_dead);
    WARN_ON(rc < 0);
    rc = 0;
out:
    return rc;
}

subsys_initcall(net_dev_init);

3.2 关键设计

  • per-CPU 收包队列: 每个 CPU 有独立的 softnet_data，避免锁竞争
  • NAPI 机制: 注册 process_backlog 作为默认的轮询处理函数
  • 软中断: NET_TX_SOFTIRQ 和 NET_RX_SOFTIRQ 是网络收发的核心中断处理机制
  • 协议类型哈希表: ptype_base[] 用于根据以太网帧的协议号分发到对应处理函数

// net/ipv4/af_inet.c
static int __init inet_init(void)
{
    struct inet_protosw *q;
    struct list_head *r;
    int rc;

    /* 检查 sk_buff 的 cb 区域大小是否足够 */
    sock_skb_cb_check_size(sizeof(struct inet_skb_parm));

    /* ========== 步骤 1: 注册传输层协议 ==========
     * proto_register() 为每个协议创建 slab 缓存
     * 例如: tcp_prot.slab 用于分配 struct tcp_sock */
    rc = proto_register(&tcp_prot, 1);
    if (rc)
        goto out;

    rc = proto_register(&udp_prot, 1);
    if (rc)
        goto out_unregister_tcp_proto;

    rc = proto_register(&raw_prot, 1);
    if (rc)
        goto out_unregister_udp_proto;

    rc = proto_register(&ping_prot, 1);
    if (rc)
        goto out_unregister_raw_proto;

    /* ========== 步骤 2: 注册 PF_INET 协议族 ==========
     * 使 socket(AF_INET, ...) 可以工作
     * inet_family_ops.create = inet_create */
    (void)sock_register(&inet_family_ops);

#ifdef CONFIG_SYSCTL
    /* 初始化静态 sysctl 参数 */
    ip_static_sysctl_init();
#endif

    /* ========== 步骤 3: 注册网络层协议处理器 ==========
     * 这些处理器由 IP 层在解析完 IP 头后调用
     * 根据 IP 头的 protocol 字段分发到对应处理器 */
    if (inet_add_protocol(&icmp_protocol, IPPROTO_ICMP) < 0)
        pr_crit("%s: Cannot add ICMP protocol\n", __func__);
    if (inet_add_protocol(&udp_protocol, IPPROTO_UDP) < 0)
        pr_crit("%s: Cannot add UDP protocol\n", __func__);
    if (inet_add_protocol(&tcp_protocol, IPPROTO_TCP) < 0)
        pr_crit("%s: Cannot add TCP protocol\n", __func__);
#ifdef CONFIG_IP_MULTICAST
    if (inet_add_protocol(&igmp_protocol, IPPROTO_IGMP) < 0)
        pr_crit("%s: Cannot add IGMP protocol\n", __func__);
#endif

    /* ========== 步骤 4: 初始化 inetsw 数组 ==========
     * inetsw[sock_type] 是链表，存放该类型 socket 的协议处理器
     * 例如: inetsw[SOCK_STREAM] 包含 tcp_prot + inet_stream_ops */
    for (r = &inetsw[0]; r < &inetsw[SOCK_MAX]; ++r)
        INIT_LIST_HEAD(r);

    /* 注册内置协议到 inetsw
     * inetsw_array 包含: SOCK_STREAM/TCP, SOCK_DGRAM/UDP, SOCK_RAW 等 */
    for (q = inetsw_array; q < &inetsw_array[INETSW_ARRAY_LEN]; ++q)
        inet_register_protosw(q);

    /* ========== 步骤 5: 初始化 ARP ==========
     * 创建 ARP 缓存、注册 ARP 包处理器 */
    arp_init();

    /* ========== 步骤 6: 初始化 IP 层 ==========
     * 包括路由表、IP 选项处理等 */
    ip_init();

    /* 初始化 IPv4 MIB 统计计数器 */
    if (init_ipv4_mibs())
        panic("%s: Cannot init ipv4 mibs\n", __func__);

    /* ========== 步骤 7: 初始化 TCP ==========
     * 创建哈希表、slab 缓存、设置内存限制 */
    tcp_init();

    /* 初始化 UDP 内存管理 */
    udp_init();

    /* 注册 UDP-Lite 协议 */
    udplite4_register();

    /* 初始化 RAW socket */
    raw_init();

    /* 初始化 PING socket */
    ping_init();

    /* ========== 步骤 8: 初始化 ICMP ==========
     * 创建内核 ICMP socket 用于发送 ICMP 消息 */
    if (icmp_init() < 0)
        panic("Failed to create the ICMP control socket.\n");

#if defined(CONFIG_IP_MROUTE)
    /* 初始化多播路由 */
    if (ip_mr_init())
        pr_crit("%s: Cannot init ipv4 mroute\n", __func__);
#endif

    /* 注册 per-netns 初始化回调 */
    if (init_inet_pernet_ops())
        pr_crit("%s: Cannot init ipv4 inet pernet ops\n", __func__);

    /* 创建 /proc/net 下的 IPv4 统计文件 */
    ipv4_proc_init();

    /* 初始化 IP 分片重组 */
    ipfrag_init();

    /* ========== 步骤 9: 注册 IP 数据包处理器 ==========
     * 这是收包路径的入口：网卡驱动 -> ip_rcv
     * 从此 IPv4 可以收包了 */
    dev_add_pack(&ip_packet_type);

    /* 初始化 IP 隧道核心 */
    ip_tunnel_core_init();

    rc = 0;
out:
    return rc;
out_unregister_raw_proto:
    proto_unregister(&raw_prot);
out_unregister_udp_proto:
    proto_unregister(&udp_prot);
out_unregister_tcp_proto:
    proto_unregister(&tcp_prot);
    goto out;
}

fs_initcall(inet_init);

  • 协议切换表: inetsw[] 数组关联 socket 类型与协议处理器

// net/ipv4/tcp.c
void __init tcp_init(void)
{
    int max_rshare, max_wshare, cnt;
    unsigned long limit;
    unsigned int i;

    /* 1. 编译时检查：确保数据结构大小符合预期 */
    BUILD_BUG_ON(TCP_MIN_SND_MSS <= MAX_TCP_OPTION_SPACE);
    BUILD_BUG_ON(sizeof(struct tcp_skb_cb) >
             sizeof_field(struct sk_buff, cb));

    /* 2. 初始化 TCP socket 分配计数器 */
    percpu_counter_init(&tcp_sockets_allocated, 0, GFP_KERNEL);

    /* 3. 设置孤儿 socket 回收定时器
     * 孤儿 socket：应用已关闭但 TCP 仍在等待 FIN-ACK 的 socket */
    timer_setup(&tcp_orphan_timer, tcp_orphan_update, TIMER_DEFERRABLE);
    mod_timer(&tcp_orphan_timer, jiffies + TCP_ORPHAN_TIMER_PERIOD);

    /* 4. 初始化 TCP 全局哈希表结构 */
    inet_hashinfo_init(&tcp_hashinfo);

    /* 5. 初始化监听端口地址哈希表（用于 SO_REUSEPORT） */
    inet_hashinfo2_init(&tcp_hashinfo, "tcp_listen_portaddr_hash",
                thash_entries, 21, 0, 64 * 1024);

    /* 6. 创建 bind bucket 的 slab 缓存 */
    tcp_hashinfo.bind_bucket_cachep =
        kmem_cache_create("tcp_bind_bucket",
                  sizeof(struct inet_bind_bucket), 0,
                  SLAB_HWCACHE_ALIGN | SLAB_PANIC | SLAB_ACCOUNT,
                  NULL);

    /* 7. 分配已建立连接哈希表 (ehash)
     * 用于快速查找已建立的 TCP 连接
     * 大小根据系统内存动态调整，每 128KB 内存一个槽位 */
    tcp_hashinfo.ehash =
        alloc_large_system_hash("TCP established",
                    sizeof(struct inet_ehash_bucket),
                    thash_entries, 17, 0,
                    NULL, &tcp_hashinfo.ehash_mask,
                    0, thash_entries ? 0 : 512 * 1024);

    /* 8. 初始化每个哈希桶的链表头 */
    for (i = 0; i <= tcp_hashinfo.ehash_mask; i++)
        INIT_HLIST_NULLS_HEAD(&tcp_hashinfo.ehash[i].chain, i);

    /* 9. 分配 ehash 的自旋锁数组 */
    if (inet_ehash_locks_alloc(&tcp_hashinfo))
        panic("TCP: failed to alloc ehash_locks");

    /* 10. 分配端口绑定哈希表 (bhash)
     * 用于管理本地端口绑定，防止端口冲突 */
    tcp_hashinfo.bhash =
        alloc_large_system_hash("TCP bind",
                    sizeof(struct inet_bind_hashbucket),
                    tcp_hashinfo.ehash_mask + 1, 17, 0,
                    &tcp_hashinfo.bhash_size, NULL, 0, 64 * 1024);
    tcp_hashinfo.bhash_size = 1U << tcp_hashinfo.bhash_size;

    /* 11. 初始化 bhash 每个桶的锁和链表 */
    for (i = 0; i < tcp_hashinfo.bhash_size; i++) {
        spin_lock_init(&tcp_hashinfo.bhash[i].lock);
        INIT_HLIST_HEAD(&tcp_hashinfo.bhash[i].chain);
    }

    /* 12. 设置最大孤儿 socket 数量 */
    cnt = tcp_hashinfo.ehash_mask + 1;
    sysctl_tcp_max_orphans = cnt / 2;

    /* 13. 初始化 TCP 内存管理 */
    tcp_init_mem();

    /* 14. 计算每 socket 的缓冲区限制 */
    limit = nr_free_buffer_pages() << (PAGE_SHIFT - 7);
    max_wshare = min(4UL*1024*1024, limit);  /* 发送缓冲区最大 4MB */
    max_rshare = min(6UL*1024*1024, limit);  /* 接收缓冲区最大 6MB */

    /* 15. 设置发送缓冲区大小 */
    init_net.ipv4.sysctl_tcp_wmem[0] = SK_MEM_QUANTUM;
    init_net.ipv4.sysctl_tcp_wmem[1] = 16*1024;        /* 默认 16KB */
    init_net.ipv4.sysctl_tcp_wmem[2] = max(64*1024, max_wshare);

    /* 16. 设置接收缓冲区大小 */
    init_net.ipv4.sysctl_tcp_rmem[0] = SK_MEM_QUANTUM;
    init_net.ipv4.sysctl_tcp_rmem[1] = 131072;          /* 默认 128KB */
    init_net.ipv4.sysctl_tcp_rmem[2] = max(131072, max_rshare);

    /* 17. 打印哈希表配置信息 */
    pr_info("Hash tables configured (established %u bind %u)\n",
        tcp_hashinfo.ehash_mask + 1, tcp_hashinfo.bhash_size);

    /* 18. 初始化 IPv4 TCP 特定部分 */
    tcp_v4_init();

    /* 19. 初始化 TCP 度量缓存 */
    tcp_metrics_init();

    /* 20. 注册 Reno 拥塞控制算法（内核默认算法） */
    BUG_ON(tcp_register_congestion_control(&tcp_reno) != 0);

    /* 21. 初始化 TCP tasklet */
    tcp_tasklet_init();

    /* 22. 初始化 MPTCP（多路径 TCP）*/
    mptcp_init();
}

关键数据结构：

数据结构用途大小计算
ehash已建立连接的哈希表每 128KB 内存一个槽位
bhash端口绑定哈希表与 ehash 大小相同
bind_bucket端口绑定桶slab 缓存分配
listen_hash监听端口哈希表每 2MB 内存一个槽位

// net/ipv4/route.c
int __init ip_rt_init(void)
{
    void *idents_hash;
    int cpu;

    /* 1. 分配 IP ID 哈希表
     * 用于生成 IP 包的 ID 字段，防止冲突 */
    idents_hash = alloc_large_system_hash("IP idents",
                          sizeof(*ip_idents) + sizeof(*ip_tstamps),
                          0, 16, HASH_ZERO,
                          NULL, &ip_idents_mask, 2048, 256*1024);
    ip_idents = idents_hash;
    prandom_bytes(ip_idents, (ip_idents_mask + 1) * sizeof(*ip_idents));
    ip_tstamps = idents_hash + (ip_idents_mask + 1) * sizeof(*ip_idents);

    /* 2. 初始化 per-CPU 的未缓存路由列表 */
    for_each_possible_cpu(cpu) {
        struct uncached_list *ul = &per_cpu(rt_uncached_list, cpu);
        INIT_LIST_HEAD(&ul->head);
        spin_lock_init(&ul->lock);
    }

    /* 3. 创建路由缓存 slab */
    ipv4_dst_ops.kmem_cachep =
        kmem_cache_create("ip_dst_cache", sizeof(struct rtable), 0,
                  SLAB_HWCACHE_ALIGN|SLAB_PANIC, NULL);

    /* blackhole 操作复用普通 dst 的 slab 缓存 */
    ipv4_dst_blackhole_ops.kmem_cachep = ipv4_dst_ops.kmem_cachep;

    /* 4. 初始化 dst 条目计数器 */
    if (dst_entries_init(&ipv4_dst_ops) < 0)
        panic("IP: failed to allocate ipv4_dst_ops counter\n");
    if (dst_entries_init(&ipv4_dst_blackhole_ops) < 0)
        panic("IP: failed to allocate ipv4_dst_blackhole_ops counter\n");

    ipv4_dst_ops.gc_thresh = ~0;
    ip_rt_max_size = INT_MAX;

    /* 5. 初始化设备地址管理 */
    devinet_init();

    /* 6. 初始化 FIB (Forwarding Information Base)
     * FIB 存储路由表，用于路由查找 */
    ip_fib_init();

    /* 7. 创建 /proc/net/rt_cache 统计文件 */
    if (ip_rt_proc_init())
        pr_err("Unable to create route proc files\n");

#ifdef CONFIG_XFRM
    /* 8. 初始化 IPsec */
    xfrm_init();
    xfrm4_init();
#endif

    /* 9. 注册路由 netlink 处理函数 */
    rtnl_register(PF_INET, RTM_GETROUTE, inet_rtm_getroute, NULL,
              RTNL_FLAG_DOIT_UNLOCKED);

#ifdef CONFIG_SYSCTL
    register_pernet_subsys(&sysctl_route_ops);
#endif
    register_pernet_subsys(&rt_genid_ops);
    register_pernet_subsys(&ipv4_inetpeer_ops);
    return 0;
}

七、关键设计思想

7.1 分层初始化

从底层基础设施到高层协议，确保依赖关系正确：

  • sock_init() 提供 socket 框架
  • net_dev_init() 提供设备层支持
  • inet_init() 注册具体协议

7.2 延迟初始化

部分工作推迟到设备注册时完成，避免空初始化：

  • 网卡驱动在 device_initcall 阶段才加载
  • 路由表在第一个网络设备注册时才创建

7.3 per-CPU 设计

收包队列、软中断等使用 per-CPU 结构，减少锁竞争：

  • softnet_data 是 per-CPU 结构
  • NET_RX_SOFTIRQ 在每个 CPU 上独立处理

7.4 哈希表优化

根据系统内存动态调整哈希表大小，平衡性能和内存：

  • ehash: 每 128KB 内存一个槽位
  • bhash: 与 ehash 大小相同
  • listen_hash: 每 2MB 内存一个槽位

7.5 协议族注册机制

通过 sock_register() 和 proto_register() 实现协议的模块化：

  • 支持多种协议族（IPv4, IPv6, Unix, Netlink 等）
  • 支持多种 socket 类型（SOCK_STREAM, SOCK_DGRAM, SOCK_RAW 等）
