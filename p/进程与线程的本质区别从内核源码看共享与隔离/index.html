1. 核心结论

Linux 内核中没有独立的"线程"数据结构——进程和线程都用同一个 task_struct 表示，走同一个 copy_process() 创建路径。唯一的区别是 clone flags，决定了各类资源是"共享引用计数+1"还是"复制一份独立副本"。

1
2
3

fork()           → clone_flags ≈ SIGCHLD（所有资源独立复制）
pthread_create() → clone_flags = CLONE_VM | CLONE_FS | CLONE_FILES | 
                                 CLONE_SIGHAND | CLONE_THREAD | ...（资源共享）

Linus Torvalds 的设计哲学：不区分进程和线程，只有"任务"(task) 和"共享程度"的差异。

// include/linux/sched.h

struct task_struct {
    pid_t               pid;            // 内核级唯一标识（每个线程不同）
    pid_t               tgid;           // 线程组 ID（同进程的线程 tgid 相同）
    struct task_struct   *group_leader;  // 线程组主线程指针

    struct mm_struct     *mm;            // 地址空间 ← CLONE_VM 控制
    struct files_struct  *files;         // 文件描述符表 ← CLONE_FILES 控制
    struct fs_struct     *fs;            // 文件系统信息 ← CLONE_FS 控制
    struct sighand_struct *sighand;      // 信号处理程序 ← CLONE_SIGHAND 控制
    struct signal_struct *signal;        // 信号队列/资源限制 ← CLONE_THREAD 控制

    void                *stack;          // 内核栈（永远独立，不可共享）
    struct sched_entity  se;             // 调度实体（永远独立）
};

关键洞察：这些字段都是指针。共享时多个 task_struct 的指针指向同一个对象（引用计数+1）；独立时各自指向自己的副本。

内核强制的依赖关系：

1
2

CLONE_THREAD → CLONE_SIGHAND → CLONE_VM
（线程组）      （共享信号处理）   （共享地址空间）

逻辑：线程组成员必须共享信号处理程序（否则信号投递语义不一致）；共享信号处理程序必须共享地址空间（信号处理函数的代码和栈在地址空间中）。

// kernel/fork.c:1690-1734

static int copy_mm(unsigned long clone_flags, struct task_struct *tsk)
{
    struct mm_struct *mm, *oldmm;
    tsk->mm = NULL;
    tsk->active_mm = NULL;

    oldmm = current->mm;
    if (!oldmm)
        return 0;  // 内核线程无用户空间 mm

    if (clone_flags & CLONE_VM) {
        // ========== 线程路径 ==========
        // 只增加引用计数，共享同一个 mm_struct
        mmget(oldmm);
        mm = oldmm;
    } else {
        // ========== 进程路径 ==========
        // dup_mm → allocate_mm + memcpy + mm_init + dup_mmap
        // dup_mmap 内部调用 copy_page_range 实现 COW
        mm = dup_mm(tsk, current->mm);
        if (!mm)
            return -ENOMEM;
    }

    tsk->mm = mm;
    tsk->active_mm = mm;
    return 0;
}

进程：dup_mm() 分配新的 mm_struct，遍历父进程所有 VMA 创建副本，调用 copy_page_range() 设置写时复制（COW）。这是 fork() 最重的操作。

线程：仅 mmget(oldmm) 增加引用计数（一条原子指令），几乎零开销。

static int copy_files(unsigned long clone_flags, struct task_struct *tsk)
{
    struct files_struct *oldf, *newf;

    oldf = current->files;
    if (clone_flags & CLONE_FILES) {
        // 线程：共享 fd 表，引用计数+1
        atomic_inc(&oldf->count);
        goto out;
    }
    // 进程：dup_fd 复制整个 fd 表
    newf = dup_fd(oldf, NR_OPEN_MAX, &error);
    tsk->files = newf;
    ...
}

影响：线程共享 fd 表意味着一个线程 close(fd) 会影响同进程所有线程。

5. 共享地址空间却拥有私有栈——线程栈的实现

理解了第 4 节的资源共享后，一个自然的问题出现了：线程共享 mm_struct（同一个虚拟地址空间），那每个线程怎么拥有自己的栈？答案是——栈的"私有"不靠内存隔离，靠的是每个线程的 RSP 寄存器指向地址空间中不同的区域。

5.1 用户栈：glibc mmap 分配 + 内核设置 RSP

流程分两步：glibc 在用户空间分配栈内存 → 内核在 copy_thread 中设置新线程的栈指针。

第一步：pthread_create() 在调 clone() 之前，先 mmap 一块匿名内存作为栈：

1
2
3
4
5

// glibc 伪代码
stack = mmap(NULL, 8MB, PROT_READ|PROT_WRITE,
             MAP_PRIVATE|MAP_ANONYMOUS|MAP_STACK, -1, 0);
clone(CLONE_VM|CLONE_THREAD|..., stack + 8MB, ...);
//                                ^^^^^^^^^^^ 栈顶（高地址端）

这段 mmap 区域成为共享 mm_struct 中的一个新 VMA。所有线程都能看到这块内存（同一地址空间），但只有该线程的 RSP 指向它。

第二步：clone() 把栈地址传入内核，最终到达 copy_thread()：

1
2

// kernel/fork.c:2619-2633
retval = copy_thread(clone_flags, args->stack, args->stack_size, p, args->tls);

1
2
3
4
5
6
7

// arch/x86/kernel/process.c:275-299

*childregs = *current_pt_regs();       // 复制父线程全部寄存器
childregs->ax = 0;                     // 子线程返回值 = 0

if (sp)
    childregs->sp = sp;                // 关键：用新栈地址覆盖 RSP

fork() 时 sp = 0，子进程继承父进程的 RSP（COW 页表保证安全）。pthread_create() 时 sp = 栈顶，新线程指向独立的 mmap 区域。

5.2 内核栈：每个线程永远独立

1
2
3

// kernel/fork.c:1044-1087 — dup_task_struct
stack = alloc_thread_stack_node(tsk, node);  // vmalloc 分配 16KB
tsk->stack = stack;

内核栈在 vmalloc 空间中（不在用户 mm_struct 里），每个线程永远独立分配。

高地址       │  主线程栈 (8MB VMA)                    │ ← 主线程 RSP
             ├──────────────────────────────────────┤

             ├──────────────────────────────────────┤

低地址       └──────────────────────────────────────┘

内核空间 vmalloc 区域（不在用户 mm 中）:
             ┌────────────────────┐
             │ 主线程内核栈 16KB   │
             │ 线程1内核栈 16KB    │  ← 每个线程 task_struct.stack
             │ 线程2内核栈 16KB    │
             └────────────────────┘

设计本质：所有线程共享地址空间，栈只是地址空间中的一段普通 VMA，通过 RSP 寄存器的不同指向实现"私有"。这也意味着线程间可以传递栈上变量的指针（同一地址空间，合法但危险）。

    p->group_leader = p;

}

进程 A (pid=100, tgid=100, group_leader=自己)

7. 大量创建线程的影响

7.1 内核栈消耗（不可 swap）

每个线程拥有独立的内核栈，x86-64 下：

1
2
3
4

// arch/x86/include/asm/page_64_types.h:15-16

#define THREAD_SIZE_ORDER  (2 + KASAN_STACK_ORDER)  // 通常为 2
#define THREAD_SIZE  (PAGE_SIZE << THREAD_SIZE_ORDER) // = 4096 << 2 = 16384 (16KB)

1000 个线程 = 16MB 纯内核栈，且这部分内存是 vmalloc 分配、不可 swap 的物理内存。10000 个线程 = 160MB。

7.2 task_struct 与调度开销

每个线程都是一个独立的 sched_entity，加入 CFS 红黑树。线程越多：

1
2
3
4
5
6
7
8
9

 ┌──────────────────────────────────────────────────┐
 │           CFS 红黑树                              │
 │                                                   │
 │     1000 个线程 = 1000 个红黑树节点                │
 │     每次 pick_next_task 代价: O(1) 取最左节点      │
 │     每次 enqueue/dequeue:    O(log N)              │
 │     但 N=1000 时上下文切换频率急剧上升             │
 │     CPU 时间大量消耗在调度器本身                    │
 └──────────────────────────────────────────────────┘
  • 时间片被切分得更细，每个线程实际运行时间缩短
  • 上下文切换频率上升 → cache/TLB 污染加剧
  • schedule() 本身的锁竞争增大（rq->lock）

7.3 PID 资源消耗

1
2
3
4
5
6

// include/linux/threads.h:28-35

#define PID_MAX_DEFAULT  0x8000              // 32768
#define PID_MAX_LIMIT    (CONFIG_BASE_SMALL ? PAGE_SIZE * 8 : \
	(sizeof(long) > 4 ? 4 * 1024 * 1024 : PID_MAX_DEFAULT))
// 64 位系统: 上限 4,194,304

每个线程消耗一个内核 PID（即使用户空间通过 getpid() 看到的是相同的 tgid）。系统全局共享 PID 空间，大量线程会挤压其他进程的 PID 分配。

7.4 信号处理锁竞争

同一线程组共享 signal_struct，信号投递需要持有 siglock：

1
2
3
4
5

信号投递路径:
  complete_signal() → lock_task_sighand() → spin_lock(siglock) → ...

线程越多 → signal_struct 上的 siglock 竞争越激烈
         → 进程级信号投递（如 SIGTERM）需要遍历线程组所有成员

7.5 VMA 膨胀

虽然线程共享 mm_struct，但每个线程的用户态栈通过 mmap 分配独立区域（默认 8MB）：

1
2
3
4

1000 个线程 = 1000 个栈 VMA = mm_struct 中 1000+ 个 vm_area_struct
→ find_vma() 红黑树查找 O(log N) 变慢
→ /proc/<pid>/maps 极长
→ fork 此进程时 dup_mmap 需要复制 1000+ 个 VMA

// kernel/fork.c:259,900-918

static int max_threads;  // 全局线程数上限

static void set_max_threads(unsigned int max_threads_suggested)
{
    u64 threads;
    unsigned long nr_pages = totalram_pages();

    // 计算方式：总内存 / (THREAD_SIZE * 8)
    // 即内核栈最多占用总内存的 1/8
    threads = div64_u64((u64) nr_pages * (u64) PAGE_SIZE,
                        (u64) THREAD_SIZE * 8UL);

    max_threads = clamp_t(u64, threads, MIN_THREADS, MAX_THREADS);
}
