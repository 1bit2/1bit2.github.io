一、核心概念区分

1.1 模式切换 vs 上下文切换

概念模式切换 (Mode Switch)上下文切换 (Context Switch)
定义用户态 ↔ 内核态进程A ↔ 进程B
是否换进程❌ 同一个进程✅ 不同进程
切换什么特权级别 (Ring 3 → Ring 0)整个执行环境
触发场景系统调用、中断、异常调用 schedule() 且选择不同进程
开销较小较大（涉及页表、TLB等）

// 文件: kernel/sched/core.c

asmlinkage __visible void __sched schedule(void)
{
    struct task_struct *tsk = current;

    // 提交异步工作
    sched_submit_work(tsk);
    
    do {
        // 禁用抢占
        preempt_disable();
        
        // 核心调度函数
        __schedule(SM_NONE);
        
        // 重新启用抢占
        sched_preempt_enable_no_resched();
    } while (need_resched());  // 循环直到不再需要重新调度
    
    sched_update_worker(tsk);
}
EXPORT_SYMBOL(schedule);

// 文件: kernel/sched/core.c

static void __sched notrace __schedule(unsigned int sched_mode)
{
    struct task_struct *prev, *next;
    unsigned long *switch_count;
    unsigned long prev_state;
    struct rq_flags rf;
    struct rq *rq;
    int cpu;

    // 获取当前 CPU 和运行队列
    cpu = smp_processor_id();
    rq = cpu_rq(cpu);
    prev = rq->curr;  // 当前正在运行的任务

    schedule_debug(prev, !!sched_mode);

    // 禁用本地中断
    local_irq_disable();
    rcu_note_context_switch(!!sched_mode);

    // 获取运行队列锁
    rq_lock(rq, &rf);
    smp_mb__after_spinlock();

    // 更新运行队列时钟
    rq->clock_update_flags <<= 1;
    update_rq_clock(rq);

    switch_count = &prev->nivcsw;
    
    // 读取进程状态
    prev_state = READ_ONCE(prev->__state);
    
    // 如果进程不是可运行状态，从队列中移除
    if (!(sched_mode & SM_MASK_PREEMPT) && prev_state) {
        if (signal_pending_state(prev_state, prev)) {
            WRITE_ONCE(prev->__state, TASK_RUNNING);
        } else {
            prev->sched_contributes_to_load =
                (prev_state & TASK_UNINTERRUPTIBLE) &&
                !(prev_state & TASK_NOLOAD) &&
                !(prev->flags & PF_FROZEN);

            deactivate_task(rq, prev, DEQUEUE_SLEEP | DEQUEUE_NOCLOCK);
            // ...
        }
        switch_count = &prev->nvcsw;
    }

    // 选择下一个要运行的任务
    next = pick_next_task(rq, prev, &rf);
    clear_tsk_need_resched(prev);
    clear_preempt_need_resched();

    // 如果选择了不同的任务，执行上下文切换
    if (likely(prev != next)) {
        rq->nr_switches++;
        
        // 更新 rq->curr
        RCU_INIT_POINTER(rq->curr, next);
        
        ++*switch_count;

        migrate_disable_switch(rq, prev);
        psi_sched_switch(prev, next, !task_on_rq_queued(prev));

        trace_sched_switch(sched_mode & SM_MASK_PREEMPT, prev, next);

        // ★★★ 执行上下文切换 ★★★
        rq = context_switch(rq, prev, next, &rf);
    } else {
        // 没有切换，释放锁
        rq->clock_update_flags &= ~(RQCF_ACT_SKIP|RQCF_REQ_SKIP);
        rq_unpin_lock(rq, &rf);
        __balance_callbacks(rq);
        raw_spin_rq_unlock_irq(rq);
    }
}

// 文件: kernel/sched/core.c

static __always_inline struct rq *
context_switch(struct rq *rq, struct task_struct *prev,
               struct task_struct *next, struct rq_flags *rf)
{
    // 准备切换
    prepare_task_switch(rq, prev, next);

    // 通知架构开始切换（用于虚拟化）
    arch_start_context_switch(prev);

    /*
     * 地址空间切换逻辑：
     * 
     * kernel -> kernel   lazy + transfer active
     *   user -> kernel   lazy + mmgrab() active
     *
     * kernel ->   user   switch + mmdrop() active
     *   user ->   user   switch
     */
    if (!next->mm) {
        // 切换到内核线程
        enter_lazy_tlb(prev->active_mm, next);
        next->active_mm = prev->active_mm;
        if (prev->mm)
            mmgrab(prev->active_mm);
        else
            prev->active_mm = NULL;
    } else {
        // 切换到用户进程
        membarrier_switch_mm(rq, prev->active_mm, next->mm);
        
        // ★ 切换页表（地址空间）★
        switch_mm_irqs_off(prev->active_mm, next->mm, next);

        if (!prev->mm) {
            rq->prev_mm = prev->active_mm;
            prev->active_mm = NULL;
        }
    }

    rq->clock_update_flags &= ~(RQCF_ACT_SKIP|RQCF_REQ_SKIP);

    prepare_lock_switch(rq, next, rf);

    // ★★★ 切换寄存器和栈 ★★★
    switch_to(prev, next, prev);
    barrier();

    // 完成切换（此时已经是新进程在执行了！）
    return finish_task_switch(prev);
}

}

// 文件: kernel/sched/core.c

static struct rq *finish_task_switch(struct task_struct *prev)
    __releases(rq->lock)
{
    struct rq *rq = this_rq();
    struct mm_struct *mm = rq->prev_mm;
    long prev_state;

    // 注意：此时 prev 可能在其他 CPU 上运行或已经被释放
    // 除了读取 prev->state，不应该再访问 prev 的其他字段

    vtime_task_switch(prev);
    perf_event_task_sched_in(prev, current);
    finish_task(prev);
    tick_nohz_task_switch();
    finish_lock_switch(rq);
    finish_arch_post_lock_switch();
    kcov_finish_switch(current);
    kmap_local_sched_in();

    fire_sched_in_preempt_notifiers(current);
    
    // 释放之前借用的 mm（如果有）
    if (mm) {
        membarrier_mm_sync_core_before_usermode(mm);
        mmdrop(mm);
    }

    // 如果前一个进程已经死亡，释放其资源
    if (unlikely(prev_state == TASK_DEAD)) {
        if (prev->sched_class->task_dead)
            prev->sched_class->task_dead(prev);

        put_task_struct_rcu_user(prev);
    }

    return rq;
}

// 文件: arch/x86/mm/tlb.c

void switch_mm_irqs_off(struct mm_struct *prev, struct mm_struct *next,
                        struct task_struct *tsk)
{
    struct mm_struct *real_prev = this_cpu_read(cpu_tlbstate.loaded_mm);
    u16 prev_asid = this_cpu_read(cpu_tlbstate.loaded_mm_asid);
    bool was_lazy = this_cpu_read(cpu_tlbstate_shared.is_lazy);
    unsigned cpu = smp_processor_id();
    u64 next_tlb_gen;
    bool need_flush;
    u16 new_asid;

    // 确保中断已禁用
    if (IS_ENABLED(CONFIG_PROVE_LOCKING))
        WARN_ON_ONCE(!irqs_disabled());

    // 验证 CR3 的正确性（调试模式）
#ifdef CONFIG_DEBUG_VM
    if (WARN_ON_ONCE(__read_cr3() != build_cr3(real_prev->pgd, prev_asid))) {
        __flush_tlb_all();
    }
#endif

    if (was_lazy)
        this_cpu_write(cpu_tlbstate_shared.is_lazy, false);

    // 如果切换到同一个 mm，可能只需要刷新
    if (real_prev == next) {
        // ... 懒惰切换优化 ...
        return;
    }

    // 应用安全缓解措施
    cond_mitigation(tsk);

    // 更新 CPU 掩码
    if (real_prev != &init_mm) {
        cpumask_clear_cpu(cpu, mm_cpumask(real_prev));
    }
    if (next != &init_mm)
        cpumask_set_cpu(cpu, mm_cpumask(next));
    
    // 获取 TLB 代数
    next_tlb_gen = atomic64_read(&next->context.tlb_gen);
    
    // 选择新的 ASID
    choose_new_asid(next, next_tlb_gen, &new_asid, &need_flush);

    // 标记正在切换
    this_cpu_write(cpu_tlbstate.loaded_mm, LOADED_MM_SWITCHING);
    barrier();

    // ★★★ 加载新的页表到 CR3 ★★★
    if (need_flush) {
        this_cpu_write(cpu_tlbstate.ctxs[new_asid].ctx_id, next->context.ctx_id);
        this_cpu_write(cpu_tlbstate.ctxs[new_asid].tlb_gen, next_tlb_gen);
        load_new_mm_cr3(next->pgd, new_asid, true);  // 刷新 TLB
        trace_tlb_flush(TLB_FLUSH_ON_TASK_SWITCH, TLB_FLUSH_ALL);
    } else {
        load_new_mm_cr3(next->pgd, new_asid, false); // 不刷新（使用 PCID）
        trace_tlb_flush(TLB_FLUSH_ON_TASK_SWITCH, 0);
    }

    barrier();

    // 更新状态
    this_cpu_write(cpu_tlbstate.loaded_mm, next);
    this_cpu_write(cpu_tlbstate.loaded_mm_asid, new_asid);

    if (next != real_prev) {
        cr4_update_pce_mm(next);
        switch_ldt(real_prev, next);  // 切换 LDT
    }
}

     * 保存 callee-saved 寄存器

    pushq   %rbx

    movq    %rsp, TASK_threadsp(%rdi)    /* 保存 prev 的栈指针 */
    movq    TASK_threadsp(%rsi), %rsp    /* 加载 next 的栈指针 */

    /* 更新栈保护金丝雀 */
    movq    TASK_stack_canary(%rsi), %rbx
    movq    %rbx, PER_CPU_VAR(fixed_percpu_data) + stack_canary_offset

    /*
     * RSB (Return Stack Buffer) 填充
     * 防止推测执行攻击

    FILL_RETURN_BUFFER %r12, RSB_CLEAR_LOOPS, X86_FEATURE_RSB_CTXSW

    popq    %r15
    popq    %r14
    popq    %r13
    popq    %r12
    popq    %rbx
    popq    %rbp

    /* 跳转到 C 函数完成剩余工作 */
    jmp     __switch_to

// 文件: arch/x86/kernel/process_64.c

__visible __notrace_funcgraph struct task_struct *
__switch_to(struct task_struct *prev_p, struct task_struct *next_p)
{
    struct thread_struct *prev = &prev_p->thread;
    struct thread_struct *next = &next_p->thread;
    struct fpu *prev_fpu = &prev->fpu;
    struct fpu *next_fpu = &next->fpu;
    int cpu = smp_processor_id();

    WARN_ON_ONCE(IS_ENABLED(CONFIG_DEBUG_ENTRY) &&
                 this_cpu_read(hardirq_stack_inuse));

    /* 1. 保存 FPU 状态 */
    if (!test_thread_flag(TIF_NEED_FPU_LOAD))
        switch_fpu_prepare(prev_fpu, cpu);

    /* 2. 保存 FS/GS 段寄存器 */
    save_fsgs(prev_p);

    /* 3. 加载新任务的 TLS 描述符 */
    load_TLS(next, cpu);

    /* 4. 结束懒惰模式（虚拟化相关）*/
    arch_end_context_switch(next_p);

    /* 5. 切换 DS 和 ES 段寄存器 */
    savesegment(es, prev->es);
    if (unlikely(next->es | prev->es))
        loadsegment(es, next->es);

    savesegment(ds, prev->ds);
    if (unlikely(next->ds | prev->ds))
        loadsegment(ds, next->ds);

    /* 6. 加载 FS/GS base */
    x86_fsgsbase_load(prev, next);

    /* 7. 加载 PKRU（内存保护键）*/
    x86_pkru_load(prev, next);

    /* 8. 更新 per-CPU 变量 */
    this_cpu_write(current_task, next_p);
    this_cpu_write(cpu_current_top_of_stack, task_top_of_stack(next_p));

    /* 9. 完成 FPU 切换 */
    switch_fpu_finish(next_fpu);

    /* 10. 更新内核栈指针 */
    update_task_stack(next_p);

    /* 11. 处理额外切换（I/O 权限位图等）*/
    switch_to_extra(prev_p, next_p);

    /* 12. AMD SYSRET 问题修复 */
    if (static_cpu_has_bug(X86_BUG_SYSRET_SS_ATTRS)) {
        unsigned short ss_sel;
        savesegment(ss, ss_sel);
        if (ss_sel != __KERNEL_DS)
            loadsegment(ss, __KERNEL_DS);
    }

    /* 返回 prev，用于 finish_task_switch() */
    return prev_p;
}

// 文件: arch/x86/include/asm/fpu/internal.h

/*
 * FPU 切换采用"懒惰"策略：
 * - switch_fpu_prepare() 保存旧状态
 * - switch_fpu_finish() 设置 TIF_NEED_FPU_LOAD 标志
 * - 实际加载在返回用户态或内核使用FPU时进行
 */

static inline void switch_fpu_prepare(struct fpu *old_fpu, int cpu)
{
    if (static_cpu_has(X86_FEATURE_FPU) && !(current->flags & PF_KTHREAD)) {
        // 保存 FPU 寄存器到内存
        save_fpregs_to_fpstate(old_fpu);
        
        // 记录上次运行的 CPU（用于懒惰恢复优化）
        old_fpu->last_cpu = cpu;

        trace_x86_fpu_regs_deactivated(old_fpu);
    }
}

static inline void switch_fpu_finish(struct fpu *new_fpu)
{
    // 设置标志，表示需要在返回用户态时加载 FPU
    if (cpu_feature_enabled(X86_FEATURE_FPU))
        set_thread_flag(TIF_NEED_FPU_LOAD);
}

十一、总结

11.1 上下文切换的本质

上下文切换的核心是栈切换：

1
2
3

// __switch_to_asm 中最关键的两行：
movq  %rsp, TASK_threadsp(%rdi)   // 保存当前栈指针到 prev->thread.sp
movq  TASK_threadsp(%rsi), %rsp   // 从 next->thread.sp 加载新栈指针

一旦栈指针切换：

  • 后续的 pop 从新栈取数据
  • ret 跳转到新栈上保存的返回地址
  • “魔法般"地切换到了另一个进程

11.2 上下文切换涉及的内容

  1. CPU 寄存器

    • 通用寄存器 (RAX-R15)
    • 栈指针 (RSP, RBP)
    • 段寄存器 (CS, SS, DS, ES, FS, GS)
    • FS/GS Base
    • 标志寄存器 (RFLAGS)

  2. FPU/SIMD 状态

    • x87 FPU 寄存器
    • SSE 寄存器 (XMM0-15)
    • AVX 寄存器 (YMM, ZMM)
    • MXCSR

  3. 地址空间

    • 页表 (CR3)
    • TLB（可能需要刷新）
    • LDT

  4. 其他硬件状态

    • TLS 描述符
    • I/O 权限位图
    • PKRU（内存保护键）
    • 调试寄存器

11.3 保存位置

信息类型保存位置
用户态寄存器内核栈顶 (pt_regs)
内核栈指针task_struct.thread.sp
内核态寄存器各自的内核栈 (inactive_task_frame)
FPU/SIMDtask_struct.thread.fpu
段寄存器task_struct.thread
页表mm_struct.pgd

11.4 一句话总结

上下文切换 = 保存当前进程的 CPU 状态到内存 + 从内存恢复另一个进程的 CPU 状态

核心技巧：通过切换栈指针 (RSP)，让 ret 指令自然跳转到新进程的执行位置
